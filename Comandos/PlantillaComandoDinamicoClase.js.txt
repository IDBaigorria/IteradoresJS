import { Comando } from '../Comando.js'; // ajuste la ruta
import { RegistroGlobal } from '../../Controlador/RegistroGlobal.js';

const ComandoDinamico = class extends Comando {
    static nombre() {
        return 'dinamico:accion';
    }

    static solo_desarrollo() {
        return false; // true si solo debe existir en desarrollo
    }

    static descripcion() {
        return 'Descripción breve del comando dinámico.';
    }

    static parametros() {
        return [
            // Definir aquí los parámetros igual que en una clase estática
        ];
    }

    static ejemplos() {
        return [
            'dinamico:accion --bandera1',
        ];
    }

    ejecutar(token, args) {
        // Acceder a los argumentos ya parseados
        // const banderas = args.banderas;
        // const opciones = args.opciones;
        // const posicionales = args.posicionales;

        const ctrl = RegistroGlobal.controlador();
        ctrl?.escribir_salida('Ejecutando comando dinámico');
        return true;
    }

    reversa() {
        return null; // o devolver una función
    }
};

// ─── REGISTRO AUTOMÁTICO ─────────────────────────────────
// RegistroGlobal decidirá la vía adecuada:
// - Si el Controlador ya está inicializado, lo registra inmediatamente.
// - Si no, lo encola para que se registre cuando el Controlador se inicie.
const ctrl = RegistroGlobal.controlador();
if (ctrl) {
    // El Controlador ya existe: registrar directamente
    ctrl.registrar_comando_desde_clase(ComandoDinamico);
} else {
    // El Controlador aún no está listo: encolar para registro diferido
    RegistroGlobal.encolar_comando(ComandoDinamico);
}
