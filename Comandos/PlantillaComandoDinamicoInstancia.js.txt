// Úsela cuando reciba una instancia que ya cumple la interfaz
// Comando (por ejemplo, generada desde un nodo o recuperada de
// un almacenamiento) y desee registrarla en el sistema.

import { Comando } from '../Comandos/index.js';  // ajuste la ruta
import { RegistroGlobal } from '../Controlador/RegistroGlobal.js'; // ajuste la ruta

/** @type {Comando} */
const comandoExistente = obtenerComandoDeAlgunaFuente(); // ejemplo

if (comandoExistente instanceof Comando) {
    // ─── Obtener el Controlador desde RegistroGlobal ─────────
    const ctrl = RegistroGlobal.controlador();

    if (ctrl) {
        // Opción A: registro inmediato (recomendada si el Controlador ya está inicializado)
        ctrl.registrar_comando_desde_instancia(comandoExistente);

        // Ejecutar (los argumentos se parsean automáticamente)
        // ctrl.ejecutar_comando(comandoExistente.constructor.nombre(), '--bandera1');
    } else {
        // Opción B: el Controlador aún no está listo.
        // Puede encolar la clase (no la instancia) para registro diferido,
        // pero perderá el estado particular de esta instancia.
        // Solo recomendado si el comando no depende de propiedades de la instancia.
        RegistroGlobal.encolar_comando(comandoExistente);
        console.warn('Controlador no inicializado. Se ha encolado la clase del comando (sin estado).');
    }
} else {
    console.error('La instancia no cumple la interfaz Comando.');
}
